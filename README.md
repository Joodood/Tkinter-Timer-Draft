    def start_countdown(self):
        if self.state == False:
            self.state = True
            self.min = self.minutes
            self.sec = self.seconds
            self.countdown()
    def pause_countdown(self):
        if self.state == True:
            self.state = False
            #print(instance)
    def show_stat(self):

        self.scrollbar = tk.Scrollbar(tk.TopLevel)

    def save(self):
